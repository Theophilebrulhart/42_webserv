//the scheme of a valid line is as follow :
//[server unique ID/name] [token] [route] [element]
//
//->[server unique ID/name] can be whatever you want.
//	It just need to be unique for each server you set up.
//
//->[token] is a set of two letters. The first one is used to
//	identify the element of the server you want to set up :
//	a -> server_names  -> std::vector<std::string>
//	b -> ports         -> std::vector<std::string>
//	c -> routes        -> std::map<std::string, t_route>
//	d -> max_body_size -> std::string
//	e -> backlog       -> std::string
//	x -> error_names   -> std::map<std::string, std::string> -> [error number] xx [error name]
//
//	Choosing 'c' will make the second letter an identifier to the route
//	element you want to setup :
//	ca -> route       -> std::string
//	cb -> methods     -> std::vector<std::string>
//	cc -> redirec     -> std::string
//	cd -> root        -> std::string
//	ce -> rep_listing -> bool
//	cf -> def_rep     -> std::string
//	cg -> cgi_script  -> std::string
//	ch -> cgi_addr    -> std::string
//This configuration file must always begin with "Bonjour!"
